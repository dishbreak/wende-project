#include <Servo.h> 
Servo myservoAz;  // create servo object to control a servo  - Azimuth
Servo myservoEl; //elevation servo object

#define MIN_PULSE_WIDTH       600     // the shortest pulse sent to a servo  
#define MAX_PULSE_WIDTH      2400     // the longest pulse sent to a servo 

int temp = 0;
int temp2 = 0;
int pos = 0;
int pos2 = 0;
 
void setup() 
{ 
  myservoAz.attach(5);  // attaches the Azimuth servo on pin 5 to the servo object 
  myservoEl.attach(6);  // attaches the Elevation servo on pin 6 to the servo object 
} 
 
void loop() 
{
  //initialize servos
  myservoAz.write(1500); //us
  myservoEl.write(1500); //us
  
  delay(1000);
 
  //Az Test 1
  for(temp = 0; temp < 30; temp += 1)
  {
    myservoAz.write(1485 + temp);
    delay(750);
    if(temp > 30) {
      temp = 0
    }
  }
  delay(5000);
}
