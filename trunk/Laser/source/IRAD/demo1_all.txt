#include <Servo.h> 
Servo myservoAz;  // create servo object to control a servo  - Azimuth
Servo myservoEl; //elevation servo object

int temp = 0;
int temp2 = 0;
int pos = 0;
int pos2 = 0;
 
void setup() 
{ 
  myservoAz.attach(5);  // attaches the Azimuth servo on pin 5 to the servo object 
  myservoEl.attach(6);  // attaches the Elevation servo on pin 6 to the servo object 
} 
 
void loop() 
{
  //initialize servos
  myservoAz.write(101);
  myservoEl.write(153);
  
  delay(10000);
 
  //Az Test 1
  for(temp = 0; temp < 4; temp += 1)
  {
    myservoAz.write(101);
    myservoEl.write(153);
    delay(1000);
    myservoAz.write(135);
    delay(1000);
    myservoAz.write(75);
    delay(4000);
  }
  
  delay(6000);
  
  //El Test 1
  for(temp = 0; temp < 4; temp += 1)
  {
    myservoAz.write(101);
    myservoEl.write(153);
    delay(1000);
    myservoEl.write(110);
    delay(1000);
    myservoEl.write(65);
    delay(4000);
  }
  
  delay(10000);
  
  //Az Test 2
  for(temp = 0; temp < 5; temp += 1)
  {
    myservoEl.write(153);
    for(pos = 0; pos < 170; pos += 5)
    {   
      myservoAz.write(pos); 
      delay(100); 
      if(pos == 0) {
         delay(500); 
      }
    }
  }
  
  delay(8000);
  
  //El Test 2
  for(temp = 0; temp < 5; temp += 1)
  {
    myservoAz.write(101);
    for(pos = 65; pos < 153; pos += 5)
    {     
      myservoEl.write(pos);
      delay(100); 
      if(pos == 0) {
        delay(500); 
      }
    }
  }
  
  delay(10000);
  
  //Sprinkler
  for(temp = 0; temp < 5; temp += 1)
  {
    if(pos2 > 90) { pos2 = 0; }
    for(pos = 0; pos < 170; pos += 5)
    {     
      myservoAz.write(pos);              // tell servo to go to position in variable 'pos' 
      delay(100); 
      if(pos == 0) {
        delay(500); 
      }
      pos2 = pos2+5;
      myservoEl.write(pos2);
    }
    
    for(pos = 170; pos > 0; pos -= 5)
    { 
      myservoAz.write(pos);              // tell servo to go to position in variable 'pos' 
      delay(100); 
      if(pos == 0) {
        delay(500); 
      }
      pos2 = pos2-5;
      myservoEl.write(pos2);
    }
  }
  
  delay(2500);
  
  myservoAz.write(101);
  myservoEl.write(153); 
}
