/*
  Blink
  Turns on an LED on for one second, then off for one second, repeatedly.
 
  This example code is in the public domain.
 */
 
#define ON 1
#define OFF 0
#define LOW 0
#define HIGH 255
#define LASER_PIN 2
#define AZ_SERVO_PIN 5
#define EL_SERVO_PIN 6

int temp = HIGH;
int pulseWidth = 20;
//a greater pulseWidth yields a slower movement
//inversely proportional
float pulseWidthHigh = 2.7; //milliseconds
float pulseWidthLow = pulseWidth - pulseWidthHigh; //ms

void setup() {                
  // initialize the digital pin as an output.
  // Pin 13 has an LED connected on most Arduino boards:
  pinMode(13, OUTPUT);     
  pinMode( LASER_PIN, OUTPUT );
  pinMode( AZ_SERVO_PIN, OUTPUT );
  pinMode( EL_SERVO_PIN, OUTPUT );
}

void loop() {
 digitalWrite(13, HIGH);   // set the LED on
  //delay(10);              // wait for a second
  //digitalWrite(13, LOW);    // set the LED off
  //delay(10);              // wait for a second
  if (temp == LOW) {
    temp = HIGH;
    analogWrite(AZ_SERVO_PIN, temp); 
    delay(pulseWidthHigh);
  }
  else {
    temp = LOW;
    analogWrite(AZ_SERVO_PIN, temp); 
    delay(pulseWidthLow);
  }  
}
